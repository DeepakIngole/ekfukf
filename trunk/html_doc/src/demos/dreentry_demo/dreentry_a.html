0015 function x = dreentry_a(xw,param)
0016 
0017   dt  = param{1};
0018   b0  = param{2};
0019   H0  = param{3};
0020   Gm0 = param{4};
0021   R0  = param{5};
0022   
0023   x = xw(1:5,:);
0024 
0025   R = sqrt(x(1,:).^2 + x(2,:).^2);
0026   V = sqrt(x(3,:).^2 + x(4,:).^2);
0027   b = b0 * exp(x(5,:));
0028   D = b .* exp((R0-R)/H0) .* V;
0029   G = -Gm0 ./ R.^3;
0030   dot_x = zeros(size(x));
0031   dot_x(1,:) = x(3,:);
0032   dot_x(2,:) = x(4,:);
0033   dot_x(3,:) = D .* x(3,:) + G .* x(1,:);
0034   dot_x(4,:) = D .* x(4,:) + G .* x(2,:);
0035   dot_x(5,:) = zeros(1,size(x,2));
0036 
0037   % Euler integration
0038   x = x + dt * dot_x;
0039 
0040   % Add process noise if the state is augmented
0041   if size(xw,1) > 5 && length(param) > 5
0042     L   = param{6};  
0043     w = xw(size(L,1)+1:size(L,1)+size(L,2),:);
0044     x = x + L * w;
0045   end
