0013 function x = reentry_f(xw,param)
0014 
0015   dt  = param{1};
0016   b0  = param{2};
0017   H0  = param{3};
0018   Gm0 = param{4};
0019   R0  = param{5};
0020   
0021   x = xw(1:5,:);
0022 
0023   R = sqrt(x(1,:).^2 + x(2,:).^2);
0024   V = sqrt(x(3,:).^2 + x(4,:).^2);
0025   b = b0 * exp(x(5,:));
0026   D = b .* exp((R0-R)/H0) .* V;
0027   G = -Gm0 ./ R.^3;
0028   dot_x = zeros(size(x));
0029   dot_x(1,:) = x(3,:);
0030   dot_x(2,:) = x(4,:);
0031   dot_x(3,:) = D .* x(3,:) + G .* x(1,:);
0032   dot_x(4,:) = D .* x(4,:) + G .* x(2,:);
0033   dot_x(5,:) = zeros(1,size(x,2));
0034 
0035   % Euler integration
0036   x = x + dt * dot_x;
0037 
0038   % Add process noise if the state is augmented
0039   if size(xw,1) > 5 && length(param) > 5
0040     L   = param{6};  
0041     w = xw(size(L,1)+1:size(L,1)+size(L,2),:);
0042     x = x + L * w;
0043   end
