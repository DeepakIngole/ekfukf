0012 function Y = bot_h(x,s)
0013   Y = zeros(size(s,2),size(x,2));
0014 
0015   for i=1:size(s,2)
0016     h = atan2(x(2,:)-s(2,i),x(1,:)-s(1,i));
0017     np = find(h>0.5*pi);
0018     nn = find(h<-0.5*pi);
